    <script>
        let cart = [];
        function toggleCart() { document.getElementById('cart-sidebar').classList.toggle('-translate-x-full'); }
        function addToCart(n, p) {
            const item = cart.find(i => i.name === n);
            if (item) item.q++; else cart.push({ name: n, price: p, q: 1 });
            updateCart();
            if (document.getElementById('cart-sidebar').classList.contains('-translate-x-full')) toggleCart();
        }
        function updateCart() {
            const container = document.getElementById('cart-items');
            let total = 0, count = 0;
            container.innerHTML = cart.map((item, i) => {
                total += item.price * item.q;
                count += item.q;
                return `<div class="flex justify-between p-3 bg-white/5 rounded-lg border border-white/5">
                    <div><h4 class="text-sm font-bold">${item.name}</h4><p class="text-xs text-purple-400">${item.price.toLocaleString()} DA × ${item.q}</p></div>
                    <button onclick="cart.splice(${i},1);updateCart();" class="text-red-400"><i class="fas fa-trash"></i></button>
                </div>`;
            }).join('');
            document.getElementById('cart-total').innerText = total.toLocaleString() + ' DA';
            document.getElementById('cart-count').innerText = count;
        }
        function openCheckout() {
            if (cart.length > 0) document.getElementById('success-modal').classList.replace('hidden', 'flex');
        }
    </script>
